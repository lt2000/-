			\begin{aligned}
			&x=\sqrt{\frac{2}{5}x^3-\frac{19}{5}x+\frac{42}{5}}\\
			&\varphi'(x)=\frac{\frac{6}{5}x^2-\frac{19}{5}}{2\sqrt{\frac{2}{5}x^3-\frac{19}{5}x+\frac{42}{5}}}\\
			&\varphi'(3)=\frac{7\sqrt{195}}{78}=1.2532>1				
			\end{aligned}
		\end{displaymath}
